pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Status code response is OK", function () {
    pm.response.to.have.status('OK');
});

var jsonData = pm.response.json();

pm.test("Request Body Property", function () {
    pm.expect(jsonData).to.be.an('object');
    pm.expect(jsonData.hasResult).to.be.a('boolean');
    pm.expect(jsonData.result).to.be.an('object');
    pm.expect(jsonData.resultType).to.be.a('number');
    pm.expect(jsonData.message).to.be.null;
    pm.expect(jsonData.validationMessages).to.be.null;
    pm.expect(jsonData.successful).to.be.a('boolean');
    pm.expect(jsonData.errorCode).to.be.a('number');
    pm.expect(jsonData.responseCode).to.be.null;
});

pm.test("hasResult is true", function () {
    pm.expect(jsonData.hasResult).to.eql(true);
});

pm.test("resultType is 1", function () {
    pm.expect(jsonData.resultType).to.eql(1);
});

pm.test("successful is true", function () {
    pm.expect(jsonData.successful).to.eql(true);
});

pm.test("errorCode is 0", function () {
    pm.expect(jsonData.errorCode).to.eql(0);
});
