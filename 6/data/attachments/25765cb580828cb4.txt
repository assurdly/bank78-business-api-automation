// Verify that the response has a 'validationMessages' array
pm.test("Response has a 'validationMessages' array", function () {
    const responseBody = pm.response.json();
    pm.expect(responseBody).to.have.property("validationMessages").that.is.an("array");
});

// Verify that the 'validationMessages' array contains the expected error message
pm.test("Validation error message is 'Account Number can only contain digits' (Expected)", function () {
    const validationMessages = pm.response.json().validationMessages;
    pm.expect(validationMessages).to.include("Account Number can only contain digits");
});

// Verify that 'successful' property is false
pm.test("'successful' property is false (Expected)", function () {
    const successful = pm.response.json().successful;
    pm.expect(successful).to.be.false;
});

// Verify that 'resultType' is 3
pm.test("Result type is 3 (Expected)", function () {
    const resultType = pm.response.json().resultType;
    pm.expect(resultType).to.equal(3);
});

// Verify that 'errorCode' property is 0
pm.test("'errorCode' property is 0 (Expected)", function () {
    const errorCode = pm.response.json().errorCode;
    pm.expect(errorCode).to.equal(0);
});

// Verify that 'responseCode' property is null
pm.test("'responseCode' property is null (Expected)", function () {
    const responseCode = pm.response.json().responseCode;
    pm.expect(responseCode).to.be.null;
});
