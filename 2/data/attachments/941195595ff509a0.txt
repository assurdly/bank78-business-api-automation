pm.test("Status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Status code is Bad Request", function () {
    pm.response.to.have.status('Bad Request');
});

var jsonData = pm.response.json();

pm.test("Response body properties", function () {
    pm.expect(jsonData).to.be.an('object');
    pm.expect(jsonData.hasResult).to.be.an('boolean');
    pm.expect(jsonData.result).to.be.null;
    pm.expect(jsonData.resultType).to.be.a('number');
    pm.expect(jsonData.message).to.be.a('string');
    pm.expect(jsonData.validationMessages).to.be.null;
    pm.expect(jsonData.successful).to.be.an('boolean');
});

pm.test("hasResult is false", function () {
    pm.expect(jsonData.hasResult).to.eql(false);
});

pm.test("resultType is 2", function () {
    pm.expect(jsonData.resultType).to.eql(2);
});

pm.test("message is present", function () {
    pm.expect(jsonData.message).to.eql("Pin and confirm pin don't match");
});

pm.test("successful is present", function () {
    pm.expect(jsonData.successful).to.eql(false);
});
