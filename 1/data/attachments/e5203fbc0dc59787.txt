pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Status code response", function () {
    pm.response.to.have.status("OK");
});

pm.test("Content-Type is present", function () {
    pm.response.to.have.header("Content-Type");
});

var jsonData = pm.response.json();

pm.test("Response body has a result property", function () {
    pm.expect(jsonData).to.be.an("object");
});

pm.test("Result property is an object", function () {
    pm.expect(jsonData.result).to.be.an("object");
    pm.expect(jsonData.result).to.have.property("message");
});

pm.test("Result property has a message property", function () {
    pm.expect(jsonData.result.message).to.be.a("string");
    pm.expect(jsonData.result.message).to.eq("PIN Created Successfully");
});

pm.test("ResultType is a number", function () {
    pm.expect(jsonData.resultType).to.be.a("number");
    pm.expect(jsonData.resultType).to.eq(1);
});

pm.test("Message is a string", function () {
    pm.expect(jsonData.message).to.be.a("string");
    pm.expect(jsonData.message).to.eql("Success");
});

pm.test("Successful is a boolean", function () {
    pm.expect(jsonData.successful).to.be.a("boolean");
    pm.expect(jsonData.successful).to.eql(true);
});
