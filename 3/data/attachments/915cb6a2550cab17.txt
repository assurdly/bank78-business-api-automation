// Verify that the response has a result object
pm.test("Response has a 'result' object", function () {
    const responseBody = pm.response.json();
    pm.expect(responseBody).to.have.property("result");
});

// Verify that the 'result' object has the expected 'message' property
pm.test("Result object has 'message' property with value 'Beneficiary successfully added'", function () {
    const result = pm.response.json().result;
    pm.expect(result).to.have.property("message", "Beneficiary successfully added");
});

// Verify that the 'resultType' is 1
pm.test("Result type is 1 (Expected)", function () {
    const resultType = pm.response.json().resultType;
    pm.expect(resultType).to.equal(1);
});
