pm.test("Status code is 401", function () {
    pm.response.to.have.status(401);
});

pm.test("Status code response is Unauthorized", function () {
    pm.response.to.have.status('Unauthorized');
});

var jsonData = pm.response.json();

pm.test("Response body property", function () {
    pm.expect(jsonData).to.be.an('object');
    pm.expect(jsonData.hasResult).to.be.a('boolean');
    pm.expect(jsonData.result).to.be.null;
    pm.expect(jsonData.resultType).to.be.a('string');
    pm.expect(jsonData.message).to.be.a('string');
    pm.expect(jsonData.validationMessages).to.be.null;
    pm.expect(jsonData.successful).to.be.a('boolean');
});

pm.test("hasResult is false", function () {
    pm.expect(jsonData.hasResult).to.eql(false);
});

pm.test("resultType is Error", function () {
    pm.expect(jsonData.resultType).to.eql('Error');
});

pm.test("message is present", function () {
    pm.expect(jsonData.message).to.eql('Sorry you cannot open an additional Account until KYC is completed');
});

pm.test("successfule is false", function () {
    pm.expect(jsonData.successful).to.eql(false);
});
