pm.test("Response code is 200 (OK)", function () {
    pm.response.to.have.status(200);
});

pm.test("Response structure and properties for 200", function () {
    pm.expect(pm.response.json()).to.have.property("hasResult", true);
    pm.expect(pm.response.json()).to.have.property("result");
    pm.expect(pm.response.json()).to.have.property("resultType", 1);
    pm.expect(pm.response.json()).to.have.property("message", null);
    pm.expect(pm.response.json()).to.have.property("validationMessages");
    pm.expect(pm.response.json()).to.have.property("successful", true);
    pm.expect(pm.response.json()).to.have.property("errorCode", 0);
    pm.expect(pm.response.json()).to.not.have.property("responseCode", "null");
});

pm.test("Result object properties for 200", function () {
    var result = pm.response.json().result;
    pm.expect(result).to.have.property("amount", 0.0);
    pm.expect(result).to.have.property("reference", null);
    pm.expect(result).to.have.property("accountNumber", null);
    pm.expect(result).to.have.property("message", "Transfer scheduled successfully");
    pm.expect(result).to.have.property("bankName", null);
    pm.expect(result).to.have.property("accountName", null);
    pm.expect(result).to.have.property("status", null);
    pm.expect(result).to.have.property("sessionId", null);
    pm.expect(result).to.have.property("senderName", null);
    pm.expect(result).to.have.property("senderAccountNumber", null);
});
