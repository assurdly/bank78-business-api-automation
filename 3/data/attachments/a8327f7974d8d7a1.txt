pm.test("Status code is 400 Bad Request", function () {
    pm.response.to.have.status(400);
});

pm.test("Response has expected structure", function () {
    pm.expect(pm.response.json()).to.have.property("hasResult", false);
    pm.expect(pm.response.json()).to.have.property("result", null);
    pm.expect(pm.response.json()).to.have.property("resultType", 2);
    pm.expect(pm.response.json()).to.have.property("message", "Invalid Transaction PIN");
    pm.expect(pm.response.json()).to.have.property("validationMessages", null);
    pm.expect(pm.response.json()).to.have.property("successful", false);
    pm.expect(pm.response.json()).to.have.property("errorCode", 1003);
    pm.expect(pm.response.json()).to.have.property("responseCode", "BANK78TF01002");
});
