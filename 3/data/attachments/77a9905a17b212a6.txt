// Test for response status code
pm.test("Response status code is 200", function () {
    pm.response.to.have.status(200);
});

// Test for response body properties
pm.test("Response body has all expected properties", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property("hasResult");
    pm.expect(jsonData).to.have.property("result");
    pm.expect(jsonData).to.have.property("resultType");
    pm.expect(jsonData).to.have.property("message");
    pm.expect(jsonData).to.have.property("validationMessages");
    pm.expect(jsonData).to.have.property("successful");
    pm.expect(jsonData).to.have.property("errorCode");
    pm.expect(jsonData).to.have.property("responseCode");
});

// Test for specific property values
pm.test("hasResult property is a boolean", function () {
    var hasResult = pm.response.json().hasResult;
    pm.expect(hasResult).to.be.a("boolean");
});

pm.test("result property is an object", function () {
    var result = pm.response.json().result;
    pm.expect(result).to.be.an("object");
});

pm.test("resultType property is a number", function () {
    var resultType = pm.response.json().resultType;
    pm.expect(resultType).to.be.a("number");
});

pm.test("Validate 'result' object with 'message'", function () {
    const responseData = pm.response.json();
    pm.expect(responseData.result).to.be.an('object');
    pm.expect(responseData.result).to.have.property('message');
    pm.expect(responseData.result.message).to.equal("Bulk process has started. Please check notification center for result");
});

pm.test("successful property is a boolean", function () {
    var successful = pm.response.json().successful;
    pm.expect(successful).to.be.a("boolean");
});

pm.test("errorCode property is a number", function () {
    var errorCode = pm.response.json().errorCode;
    pm.expect(errorCode).to.be.a("number");
});

pm.test("responseCode property is null", function () {
    var responseCode = pm.response.json().responseCode;
    pm.expect(responseCode).to.be.null;
});
